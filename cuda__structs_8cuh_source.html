44/** A structure for packing neighbor index and distance for kNN search. */
45template<typename F>
46struct knn_entry
47{
48 F distance;
49 uint32_t index;
50
51 CUDA_CALLABLE_MEMBER bool operator<(const knn_entry &rhs) const
52 {
53 return distance < rhs.distance ||
54 (distance == rhs.distance && index < rhs.index);
55 }
56
57 CUDA_CALLABLE_MEMBER bool operator>(const knn_entry &rhs) const
58 {
59 return rhs < *this;
60 }
61
62 CUDA_CALLABLE_MEMBER bool operator<=(const knn_entry &rhs) const
63 {
64 return !(rhs < *this);
65 }
66
67 CUDA_CALLABLE_MEMBER bool operator>=(const knn_entry &rhs) const
68 {
69 return !(*this < rhs);
70 }
71};
