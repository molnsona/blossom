23 void
24 RawDataStats::update(const DataModel &dm)
25 {
26  if (!dirty(dm))
27  return;
28 
29  means.clear();
30  means.resize(dm.d, 0);
31  sds = means;
32 
33  size_t d = dm.d;
34  for (size_t ni = 0; ni < dm.n; ++ni) {
35  for (size_t di = 0; di < d; ++di) {
36  float tmp = dm.data[ni * d + di];
37  means[di] += tmp;
38  sds[di] += tmp * tmp;
39  }
40  }
41 
42  for (size_t di = 0; di < d; ++di) {
43  means[di] /= dm.n;
44  sds[di] /= dm.n;
45  sds[di] = sqrt(sds[di] - means[di] * means[di]);
46  }
47 
48  clean(dm);
49  touch();
50 }

52 void
53 TransData::update(const DataModel &dm, const RawDataStats &s)
54 {
55  if (dirty(dm)) {
56  config.resize(dm.d);
57  n = dm.n;
58  data.clear(); // TODO this needs to be updated if rolling stats should
59  // work
60  data.resize(n * dim(), 0);
61  sums.clear();
62  sums.resize(dim(), 0);
63  sqsums = sums;
64  touch();
65  clean(dm);
66  }
67 
68  if (stat_watch.dirty(s)) {
69  refresh(dm);
70  stat_watch.clean(s);
71  }
72 
73  // make sure we're the right size
74  auto [ri, rn] = dirty_range(dm);
75  if (!rn)
76  return;
77  // TODO: make this constant configurable (and much bigger)
78  if (rn > 10000)
79  rn = 10000;
80 
81  clean_range(dm, rn);
82  const size_t d = dim();
83  std::vector<float> sums_adjust(d, 0), sqsums_adjust(d, 0);
84 
85  for (; rn-- > 0; ++ri) {
86  if (ri >= n)
87  ri = 0;
88  for (size_t di = 0; di < d; ++di) {
89  const auto &c = config[di];
90 
91  float tmp = data[ri * d + di];
92  sums_adjust[di] -= tmp;
93  sqsums_adjust[di] -= tmp * tmp;
94  tmp = dm.data[ri * d + di];
95 
96  tmp += c.affine_adjust;
97  if (c.asinh)
98  tmp = asinhf(tmp / c.asinh_cofactor);
99 
100  data[ri * d + di] = tmp;
101  sums_adjust[di] += tmp;
102  sqsums_adjust[di] += tmp * tmp;
103  }
104  }
105 
106  for (size_t di = 0; di < d; ++di) {
107  sums[di] += sums_adjust[di];
108  sqsums[di] += sqsums_adjust[di];
109  }
110 
111  touch();
112 }
