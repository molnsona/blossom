23void
24RawDataStats::update(const DataModel &dm)
25{
26 if (!dirty(dm))
27 return;
28
29 means.clear();
30 means.resize(dm.d, 0);
31 sds = means;
32
33 size_t d = dm.d;
34 for (size_t ni = 0; ni < dm.n; ++ni) {
35 for (size_t di = 0; di < d; ++di) {
36 float tmp = dm.data[ni * d + di];
37 means[di] += tmp;
38 sds[di] += tmp * tmp;
39 }
40 }
41
42 for (size_t di = 0; di < d; ++di) {
43 means[di] /= dm.n;
44 sds[di] /= dm.n;
45 sds[di] = sqrt(sds[di] - means[di] * means[di]);
46 }
47
48 clean(dm);
49 touch();
50}

52void
53TransData::update(const DataModel &dm, const RawDataStats &s)
54{
55 if (dirty(dm)) {
56 config.resize(dm.d);
57 n = dm.n;
58 data.clear(); // TODO this needs to be updated if rolling stats should
59 // work
60 data.resize(n * dim(), 0);
61 sums.clear();
62 sums.resize(dim(), 0);
63 sqsums = sums;
64 touch();
65 clean(dm);
66 }
67
68 if (stat_watch.dirty(s)) {
69 refresh(dm);
70 stat_watch.clean(s);
71 }
72
73 const size_t max_points =
74#ifndef ENABLE_CUDA
75 10000
76#else
77 50000
78#endif
79 ;
80
81 // make sure we're the right size
82 auto [ri, rn] = dirty_range(dm);
83 if (!rn)
84 return;
85 if (rn > max_points)
86 rn = max_points;
87
88 clean_range(dm, rn);
89 const size_t d = dim();
90 std::vector<float> sums_adjust(d, 0), sqsums_adjust(d, 0);
91
92 for (; rn-- > 0; ++ri) {
93 if (ri >= n)
94 ri = 0;
95 for (size_t di = 0; di < d; ++di) {
96 const auto &c = config[di];
97
98 float tmp = data[ri * d + di];
99 sums_adjust[di] -= tmp;
100 sqsums_adjust[di] -= tmp * tmp;
101 tmp = dm.data[ri * d + di];
102
103 tmp += c.affine_adjust;
104 if (c.asinh)
105 tmp = asinhf(tmp / c.asinh_cofactor);
106
107 data[ri * d + di] = tmp;
108 sums_adjust[di] += tmp;
109 sqsums_adjust[di] += tmp * tmp;
110 }
111 }
112
113 for (size_t di = 0; di < d; ++di) {
114 sums[di] += sums_adjust[di];
115 sqsums[di] += sqsums_adjust[di];
116 }
117
118 touch();
119}
