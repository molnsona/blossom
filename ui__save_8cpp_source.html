44void
45UiSaver::render(State &state, ImGuiWindowFlags window_flags)
46{
47 if (!show_window)
48 return;
49
50 saver.Display();
51
52 if (saver.HasSelected()) {
53 try {
54 save_data(state, saver.GetSelected().string());
55 } catch (std::exception &e) {
56 saving_error = e.what();
57 }
58
59 saver.ClearSelected();
60 }
61
62 if (!saving_error.empty()) {
63 ImGui::Begin("Saving error", nullptr, window_flags);
64 ImGui::Text(saving_error.c_str());
65 if (ImGui::Button("OK"))
66 saving_error = "";
67 ImGui::End();
68 }
69
70 if (ImGui::Begin("Save##window", &show_window, window_flags)) {
71 auto save_line = [&](const char *text, int type) {
72 ImGui::Text(text);
73 std::string checkbox_name =
74 "##save_checkbox" + std::to_string(type);
75 ImGui::Checkbox(checkbox_name.data(), &data_flags[type]);
76 ImGui::SameLine();
77 std::string inputtext_name =
78 "file name##save" + std::to_string(type);
79 ImGui::InputText(inputtext_name.data(), &file_names[type]);
80 };
81
82 save_line("Save hidim points:", UiSaver::Types::POINTS_HD);
83
84 save_line("Save hidim landmarks:", UiSaver::Types::LAND_HD);
85
86 save_line("Save 2D points:", UiSaver::Types::POINTS_2D);
87
88 save_line("Save 2D landmarks:", UiSaver::Types::LAND_2D);
89
90 ImGui::NewLine();
91
92 if (ImGui::Checkbox("Save all", &all))
93 if (all)
94 std::fill(data_flags.begin(), data_flags.end(), true);
95 else
96 std::fill(data_flags.begin(), data_flags.end(), false);
97
98 ImGui::NewLine();
99
100 if (ImGui::Button("Save##button"))
101 saver.Open();
102
103 ImGui::End();
104 }
105}

123/**
124 * @brief Writes multi-dimensional data into the file by a given handler.
125 *
126 * @param dim Dimension of the data.
127 * @param data Multi-dimensional data to be written.
128 * @param handle Handle to the opened file for writing.
129 */
130static void
131write_data_float(size_t dim,
132 const std::vector<float> &data,
133 std::ofstream &handle)
134{
135 for (size_t i = 0; i < data.size(); i += dim) {
136 for (size_t j = 0; j < dim - 1; ++j) {
137 handle << data[i + j] << '\t';
138 }
139 handle << data[i + dim - 1] << '\n';
140 }
141};
142
143/**
144 * @brief Writes two-dimensional data into the file by a given handler.
145 *
146 * @param data Two-dimensional data to be written.
147 * @param handle Handle to the opened file for writing.
148 */
149static void
150write_data_2d(const std::vector<glm::vec2> &data, std::ofstream &handle)
151{
152 for (size_t i = 0; i < data.size(); ++i) {
153 handle << data[i].x << '\t' << data[i].y << '\n';
154 }
155};

157void
158UiSaver::write(Types type,
159 const State &state,
160 const std::string &dir_name) const
161{
162 std::string path = dir_name + "/" + file_names[type];
163 std::ofstream handle(path, std::ios::out);
164 if (!handle)
165 throw std::domain_error("Can not open file");
166
167 switch (type) {
168 case UiSaver::Types::POINTS_HD:
169 write_data_float(state.scaled.dim(), state.scaled.data, handle);
170 break;
171 case UiSaver::Types::LAND_HD:
172 write_data_float(
173 state.landmarks.d, state.landmarks.hidim_vertices, handle);
174 break;
175 case UiSaver::Types::POINTS_2D:
176 write_data_2d(state.scatter.points, handle);
177 break;
178 case UiSaver::Types::LAND_2D:
179 write_data_2d(state.landmarks.lodim_vertices, handle);
180 break;
181 }
182
183 handle.close();
184}
