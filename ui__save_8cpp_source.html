118 /**
119  * @brief Writes multi-dimensional data into the file by a given handler.
120  *
121  * @param dim Dimension of the data.
122  * @param data Multi-dimensional data to be written.
123  * @param handle Handle to the opened file for writing.
124  */
125 static void
126 write_data_float(size_t dim,
127  const std::vector<float> &data,
128  std::ofstream &handle)
129 {
130  for (size_t i = 0; i < data.size(); i += dim) {
131  for (size_t j = 0; j < dim - 1; ++j) {
132  handle << data[i + j] << '\t';
133  }
134  handle << data[i + dim - 1] << '\n';
135  }
136 };
137 
138 /**
139  * @brief Writes two-dimensional data into the file by a given handler.
140  *
141  * @param data Two-dimensional data to be written.
142  * @param handle Handle to the opened file for writing.
143  */
144 static void
145 write_data_2d(const std::vector<Magnum::Vector2> &data, std::ofstream &handle)
146 {
147  for (size_t i = 0; i < data.size(); ++i) {
148  handle << data[i].x() << '\t' << data[i].y() << '\n';
149  }
150 };

152 void
153 uiSaver::write(Types type, const State &state, const std::string &dir_name) const
154 {
155  std::string path = dir_name + "/" + file_names[type];
156  std::ofstream handle(path, std::ios::out);
157  if (!handle)
158  throw std::domain_error("Can not open file");
159 
160  switch (type) {
161  case uiSaver::Types::POINTS_HD:
162  write_data_float(state.scaled.dim(), state.scaled.data, handle);
163  break;
164  case uiSaver::Types::LAND_HD:
165  write_data_float(
166  state.landmarks.d, state.landmarks.hidim_vertices, handle);
167  break;
168  case uiSaver::Types::POINTS_2D:
169  write_data_2d(state.scatter.points, handle);
170  break;
171  case uiSaver::Types::LAND_2D:
172  write_data_2d(state.landmarks.lodim_vertices, handle);
173  break;
174  }
175 
176  handle.close();
177 }
