48 {
49  // TODO cache these allocations in GraphRenderer object
50  // std::vector<Vector2> vertices(model.lodim_vertices.size());
51  if (vertices.size() != model.lodim_vertices.size()) {
52  vertices.clear();
53  vertices.resize(model.lodim_vertices.size());
54  }
55 
56  for (size_t i = 0; i < vertices.size(); ++i) {
57  vertices[i] = view.screen_coords(model.lodim_vertices[i]);
58  }
59 
60  std::vector<Vector2> edge_lines(2 * model.edges.size());
61  for (size_t i = 0; i < model.edges.size(); ++i) {
62  edge_lines[2 * i + 0] = vertices[model.edges[i].first];
63  edge_lines[2 * i + 1] = vertices[model.edges[i].second];
64  }
65 
66  GL::Buffer buffer;
67  buffer.setData(
68  Corrade::Containers::ArrayView(edge_lines.data(), edge_lines.size()));
69 
70  GL::Renderer::enable(GL::Renderer::Feature::Blending);
71 
72  line_mesh.setCount(edge_lines.size())
73  .addVertexBuffer(std::move(buffer), 0, decltype(flat_shader)::Position{});
74 
75  auto screen_proj = view.screen_projection_matrix();
76 
77  flat_shader.setTransformationProjectionMatrix(screen_proj)
78  .setColor(0xc01010_rgbf)
79  .draw(line_mesh);
80 
81  flat_shader.setColor(0x66666666_rgbaf);
82  for (auto &&v : vertices) {
83  flat_shader
84  .setTransformationProjectionMatrix(
85  screen_proj * Matrix3::translation(v) *
86  Matrix3::scaling(Vector2(vertex_size)))
87  .draw(circle_mesh);
88  }
89 
90  GL::Renderer::disable(GL::Renderer::Feature::Blending);

93 bool
94 GraphRenderer::is_vert_pressed(Magnum::Vector2 mouse, size_t &vert_ind) const
95 {
96  float radius = vertex_size;
97 
98  for (size_t i = 0; i < vertices.size(); ++i) {
99  auto vert = vertices[i];
100  if ((mouse.x() >= roundf(vert.x()) - radius) &&
101  (mouse.x() <= roundf(vert.x()) + radius) &&
102  (mouse.y() >= roundf(vert.y()) - radius) &&
103  (mouse.y() <= roundf(vert.y()) + radius)) {
104  vert_ind = i;
105  return true;
106  }
107  }
108 
109  return false;
