45void
46GraphRenderer::draw(const View &view, const LandmarkModel &model)
47{
48 glEnable(GL_BLEND);
49
50 prepare_data(view.current_zoom, model);
51
52 shader_e.use();
53 shader_e.set_mat4("model", glm::mat4(1.0f));
54 shader_e.set_mat4("view", view.get_view_matrix());
55 shader_e.set_mat4("proj", view.get_proj_matrix());
56
57 glBindVertexArray(VAO_e);
58 glDrawArrays(GL_LINES, 0, 2 * model.edges.size());
59
60 shader_v.use();
61 shader_v.set_mat4("model", glm::mat4(1.0f));
62 shader_v.set_mat4("view", view.get_view_matrix());
63 shader_v.set_mat4("proj", view.get_proj_matrix());
64
65 glBindVertexArray(VAO_v);
66
67 for (size_t i = 0; i < model.lodim_vertices.size(); ++i) {
68 glDrawArrays(GL_TRIANGLE_FAN, i * num_all_vtxs, num_all_vtxs);
69 }
70
71 glDisable(GL_BLEND);
72}

81void
82GraphRenderer::add_circle(float middle_x,
83 float middle_y,
84 float zoom,
85 std::vector<float> &all_vtxs)
86{
87 int sides = 12;
88 float radius = 0.05f;
89 num_all_vtxs = sides + 2;
90
91 double two_pi = 2.0f * M_PI;
92
93 all_vtxs.emplace_back(middle_x);
94 all_vtxs.emplace_back(middle_y);
95
96 for (int i = 1; i < num_all_vtxs; i++) {
97 all_vtxs.emplace_back(middle_x +
98 (radius * cos(i * two_pi / sides)) * zoom * 130);
99 all_vtxs.emplace_back(middle_y +
100 (radius * sin(i * two_pi / sides)) * zoom * 130);
101 }
102}

104void
105GraphRenderer::prepare_vertices(float current_zoom, const LandmarkModel &model)
106{
107 if (vertices.size() != model.lodim_vertices.size()) {
108 vertices.clear();
109 vertices.resize(model.lodim_vertices.size());
110 }
111
112 std::vector<float> all_vtxs;
113
114 for (size_t i = 0; i < vertices.size(); ++i) {
115 vertices[i] = model.lodim_vertices[i];
116 add_circle(vertices[i].x, vertices[i].y, current_zoom, all_vtxs);
117 }
118
119 glBindVertexArray(VAO_v);
120
121 glBindBuffer(GL_ARRAY_BUFFER, VBO_v);
122 glBufferData(GL_ARRAY_BUFFER,
123 all_vtxs.size() * sizeof(float),
124 &all_vtxs[0],
125 GL_DYNAMIC_DRAW);
126 glVertexAttribPointer(
127 0, 2, GL_FLOAT, GL_FALSE, 2 * sizeof(float), (void *)0);
128 glEnableVertexAttribArray(0);
129}

131void
132GraphRenderer::prepare_edges(const LandmarkModel &model)
133{
134 std::vector<glm::vec2> edge_lines(2 * model.edges.size());
135 for (size_t i = 0; i < model.edges.size(); ++i) {
136 edge_lines[2 * i + 0] = vertices[model.edges[i].first];
137 edge_lines[2 * i + 1] = vertices[model.edges[i].second];
138 }
139
140 glBindVertexArray(VAO_e);
141
142 glBindBuffer(GL_ARRAY_BUFFER, VBO_e);
143 glBufferData(GL_ARRAY_BUFFER,
144 edge_lines.size() * sizeof(glm::vec2),
145 &edge_lines[0],
146 GL_DYNAMIC_DRAW);
147 glVertexAttribPointer(
148 0, 2, GL_FLOAT, GL_FALSE, sizeof(glm::vec2), (void *)0);
149 glEnableVertexAttribArray(0);
150}

152bool
153GraphRenderer::is_vert_pressed(const View &view,
154 glm::vec2 mouse,
155 size_t &vert_ind) const
156{
157 float radius = vertex_size;
158
159 for (size_t i = 0; i < vertices.size(); ++i) {
160 glm::vec2 vert = view.screen_coords(vertices[i]);
161
162 if ((mouse.x >= roundf(vert.x) - radius) &&
163 (mouse.x <= roundf(vert.x) + radius) &&
164 (mouse.y >= roundf(vert.y) - radius) &&
165 (mouse.y <= roundf(vert.y) + radius)) {
166 vert_ind = i;
167 return true;
168 }
169 }
170
171 return false;
172}
