28 /**
29  * @brief Converts hsv color to rgb color system.
30  *
31  * @param h Hue
32  * @param s Saturation
33  * @param v Value
34  * @return std::tuple<uint8_t, uint8_t, uint8_t> Color in the RGB color system.
35  */
36 static std::tuple<uint8_t, uint8_t, uint8_t>
37 hsv2rgb(float h, float s, float v)
38 {
39  float chroma = v * s;
40  float m = v - chroma;
41  h *= 6;
42  int hi = truncf(h);
43  float rest = h - hi;
44  hi %= 6;
45 
46  float rgb[3] = { 0, 0, 0 };
47  switch (hi) {
48  case 0:
49  rgb[0] = 1;
50  rgb[1] = rest;
51  break;
52  case 1:
53  rgb[1] = 1;
54  rgb[0] = 1 - rest;
55  break;
56  case 2:
57  rgb[1] = 1;
58  rgb[2] = rest;
59  break;
60  case 3:
61  rgb[2] = 1;
62  rgb[1] = 1 - rest;
63  break;
64  case 4:
65  rgb[2] = 1;
66  rgb[0] = rest;
67  break;
68  case 5:
69  default:
70  rgb[0] = 1;
71  rgb[2] = 1 - rest;
72  break;
73  }
74 
75  for (size_t i = 0; i < 3; ++i)
76  rgb[i] = (chroma * rgb[i] + m) * 255;
77  return { rgb[0], rgb[1], rgb[2] };
78 }

98 void
99 ColorData::update(const TransData &td)
100 {
101  if (td.n != data.size()) {
102  data.resize(td.n, Magnum::Color4(0, 0, 0, 0));
103  refresh(td);
104  }
105 
106  auto [ri, rn] = dirty_range(td);
107  if (!rn)
108  return;
109  if (rn > 1000)
110  rn = 1000;
111 
112  size_t n = td.n;
113  size_t d = td.dim();
114 
115  clean_range(td, rn);
116  switch (coloring) {
117  case (int)ColorData::Coloring::EXPR: {
118  if (expr_col >= d)
119  expr_col = 0;
120 
121  auto pal = colormap::palettes.at(col_palette).rescale(0, 1);
122  float mean = td.sums[expr_col] / n;
123  float sdev = sqrt(td.sqsums[expr_col] / n - mean * mean);
124 
125  for (; rn-- > 0; ++ri) {
126  if (ri >= n)
127  ri = 0;
128  auto c =
129  reverse
130  ? pal(1 - pnormf(td.data[ri * d + expr_col], mean, sdev))
131  : pal(pnormf(td.data[ri * d + expr_col], mean, sdev));
132  data[ri] = Magnum::Color4(c.channels[0].val / 255.0f,
133  c.channels[1].val / 255.0f,
134  c.channels[2].val / 255.0f,
135  alpha);
136  }
137  } break;
138 
139  case int(ColorData::Coloring::CLUSTER):
140  if (cluster_col >= d)
141  cluster_col = 0;
142 
143  std::vector<std::tuple<unsigned char, unsigned char, unsigned char>>
144  pal;
145 
146  create_col_palette(cluster_cnt, pal);
147 
148  for (; rn-- > 0; ++ri) {
149  if (ri >= n)
150  ri = 0;
151 
152  auto cluster = td.data[ri * d + cluster_col];
153 
154  auto [r, g, b] =
155  std::isnan(cluster)
156  ? std::make_tuple<unsigned char,
157  unsigned char,
158  unsigned char>(128, 128, 128)
159  : pal[(int)roundf(cluster) % (cluster_cnt)];
160  data[ri] =
161  Magnum::Color4(r / 255.0f, g / 255.0f, b / 255.0f, alpha);
162  }
163  break;
164  }
165 }
