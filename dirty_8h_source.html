25 /** A piece of dirt for dirtying the caches.
26  *
27  * This is supposed to be a part of classes that are cached elsewhere; having
28  * the `dirty` counter allows us to reliably observe if someting changed and
29  * run the updates accordingly. Use with `Cleaner` to observe the changes.
30  */
31 struct Dirt
32 {
33  int dirt;
34 
35  Dirt()
36  : dirt(0)
37  {}
38  /** Make the cache dirty
39  *
40  * Call this if something changed and the caches need to be refreshed
41  */
42  void touch() { ++dirt; }
43 };
44 
45 /** A piece of cache that keeps track of the dirty status.
46  *
47  * This is a natural counterpart of the `Dirt` class.
48  */
49 struct Cleaner
50 {
51  int cleaned;
52 
53  Cleaner()
54  : cleaned(-1)
55  {}
56 
57  /** Returns true if the cache needs to be refreshed */
58  bool dirty(const Dirt &d)
59  {
60  // handle overflows!
61  return d.dirt - cleaned > 0;
62  }
63 
64  /** Call this when the cache is refreshed. */
65  void clean(const Dirt &d)
66  {
67  if (cleaned < d.dirt)
68  cleaned = d.dirt;
69  }
70 };

72 /** Multi-piece cache-dirtying object.
73  *
74  * This is to be inherited into data objects that are cached elsewhere, but
75  * have more parts that may be transformed independently. Use with `Sweeper`.
76  *
77  * User structures are responsible for filling in `n` correctly.
78  */

87 /** A piece of multi-object cache.
88  *
89  * This is the counterpart of `Dirts`.
90  */
91 struct Sweeper : public Cleaner
92 {
93  size_t begin, dirts;
94 
95  Sweeper()
96  : begin(0)
97  , dirts(0)
98  {}
99 
100  /** Force-refresh the whole range */
101  void refresh(const Dirts &d) { dirts = d.n; }
102  void refresh(size_t n_dirts) { dirts = n_dirts; }
103 
104  /** Find the range to refresh
105  *
106  * Return the beginning index and size of the range that needs to be
107  * refreshed. Note that the range is cyclic!
108  */
109  std::tuple<size_t, size_t> dirty_range(const Dirts &d)
110  {
111  if (dirty(d)) {
112  refresh(d);
113  clean(d);
114  }
115  if (begin >= d.n)
116  begin = 0;
117  return { begin, dirts };
118  }
119 
120  /** Clean a range of the cache
121  *
122  * Call this to tell the Sweeper that you refresh `n` cache elements,
123  * starging at the beginning index returned from dirty_range().
124  * The indexing is cyclic modulo `d.n`.
125  */
126  void clean_range(const Dirts &d, size_t n)
127  {
128  if (n > dirts)
129  n = dirts;
130  dirts -= n;
131  if (!dirts)
132  begin = 0;
133  else {
134  begin += n;
135  if (begin >= d.n)
136  begin -= d.n;
137  }
138  }
139 };
