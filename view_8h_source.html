35/**
36 * @brief A small utility class that manages the viewport coordinates, together
37 * with the virtual "camera" position and zoom.
38 */
39class View
40{
41public:
42 // Default View values
43 const float movement_speed = 0.2f;
44 const float smooth_speed = 0.005f;
45 const float zoom_depth = 0.009f;
46
47 // View vectors
48 glm::vec3 target_pos;
49 glm::vec3 current_pos;
50 glm::vec3 front;
51 glm::vec3 up;
52 glm::vec3 right;
53 glm::vec3 world_up;
54
55 float target_zoom;
56 float current_zoom;
57
58 // Framebuffer size
59 int width;
60 int height;
61
62 View(int width = 800,
63 int height = 600,
64 glm::vec3 position = glm::vec3(0.0f, 0.0f, 1.0f),
65 glm::vec3 up = glm::vec3(0.0f, 1.0f, 0.0f))
66 : front(glm::vec3(0.0f, 0.0f, -1.0f))
67 , target_zoom(zoom_depth)
68 , current_zoom(zoom_depth)
69 {
70 target_pos = position;
71 current_pos = position;
72 world_up = up;
73 update_view_vectors();
74 }
75
76 /**
77 * @brief Move the current position and zoom a bit closer to the target
78 * position and zoom.
79 *
80 * @param dt Time difference.
81 * @param callbacks Values collected by event callbacks.
82 */
83 void update(float dt, const CallbackValues &callbacks)
84 {
85 width = callbacks.fb_width;
86 height = callbacks.fb_height;
87
88 process_keyboard(callbacks.key, callbacks.key_action);
89
90 process_mouse_scroll(callbacks.yoffset,
91 glm::vec2(callbacks.xpos, callbacks.ypos));
92
93 process_mouse_button(callbacks.button,
94 callbacks.mouse_action,
95 glm::vec2(callbacks.xpos, callbacks.ypos));
96
97 float power = pow(smooth_speed, dt);
98 float r = 1 - power;
99 current_pos = power * current_pos + r * target_pos;
100 current_zoom = power * current_zoom + r * target_zoom;
101
102 update_view_vectors();
103 }
104
105 /**
106 * @brief Compute view matrix for drawing into the "view" space.
107 *
108 * @return glm::mat4 View matrix
109 */
110 glm::mat4 get_view_matrix() const
111 {
112 return glm::lookAt(current_pos, current_pos + front, up);
113 }
114
115 /**
116 * @brief Compute projection matrix for orthographic projection.
117 *
118 * @return glm::mat4 Projection matrix
119 */
120 glm::mat4 get_proj_matrix() const
121 {
122 float half_w = width / 2.0f;
123 float half_h = height / 2.0f;
124 return glm::ortho(-half_w * current_zoom,
125 half_w * current_zoom,
126 -half_h * current_zoom,
127 half_h * current_zoom,
128 0.1f,
129 100.0f);
130 }
131
132 /**
133 * @brief Convert mouse coordinates ([0,0] in the upper left corner),
134 * to screen coordinates ([0,0] in the middle of the screen).
135 *
136 * @param mouse Mouse coordinates ([0,0] in the upper left corner)
137 * @return glm::vec2 Screen coordinates ([0,0] in the middle of the screen)
138 */
139 glm::vec2 screen_mouse_coords(glm::vec2 mouse) const
140 {
141 return glm::vec2(mouse.x - width / 2.0f, height / 2.0f - mouse.y);
142 }
143
144 /**
145 * @brief Convert mouse coordinates ([0,0] in the upper left corner),
146 * to model coordinates ([0,0] in the middle of the screen).
147 *
148 * @param mouse Mouse coordinates ([0,0] in the upper left corner)
149 * @return glm::vec2 Model coordinates ([0,0] in the middle of the screen)
150 */
151 glm::vec2 model_mouse_coords(glm::vec2 mouse) const
152 {
153 glm::vec3 res =
154 glm::unProject(glm::vec3(mouse.x, height - mouse.y, 0.1f),
155 get_view_matrix(),
156 get_proj_matrix(),
157 glm::vec4(0.0f, 0.0f, width, height));
158
159 return glm::vec2(res.x, res.y);
160 }
161
162 /**
163 * @brief Converts point to screen coordinates([0,0] in the middle of the
164 * screen).
165 *
166 * @param point Input point in original not projected coordinates.
167 * @return glm::vec2 Point in screen coordinates.
168 */
169 glm::vec2 screen_coords(glm::vec2 point) const
170 {
171 glm::vec3 res = glm::project(glm::vec3(point, 0.1f),
172 get_view_matrix(),
173 get_proj_matrix(),
174 glm::vec4(0.0f, 0.0f, width, height));
175 return screen_point_coords(glm::vec2(res.x, res.y));
176 }
177
178private:
179 /**
180 * @brief Re-calculates the right and up vector from the View's updated
181 * vectors.
182 *
183 */
184 void update_view_vectors()
185 {
186 right = glm::normalize(glm::cross(
187 front, world_up)); // normalize the vectors, because their length gets
188 // closer to 0 the more you look up or down which
189 // results in slower movement.
190 up = glm::normalize(glm::cross(right, front));
191 }
192
193 /**
194 * @brief Process keyboard input and move the "camera" accordingly.
195 *
196 * @param key Name of the used key.
197 * @param action Pressed, released or hold key.
198 */
199 void process_keyboard(int key, int action)
200 {
201 float half_h = height / 2.0f;
202 float velocity = half_h * target_zoom * movement_speed;
203 if (key == GLFW_KEY_W &&
204 (action == GLFW_PRESS || action == GLFW_REPEAT))
205 target_pos.y += velocity;
206 if (key == GLFW_KEY_S &&
207 (action == GLFW_PRESS || action == GLFW_REPEAT))
208 target_pos.y -= velocity;
209 if (key == GLFW_KEY_A &&
210 (action == GLFW_PRESS || action == GLFW_REPEAT))
211 target_pos.x -= velocity;
212 if (key == GLFW_KEY_D &&
213 (action == GLFW_PRESS || action == GLFW_REPEAT))
214 target_pos.x += velocity;
215 }
216
217 /**
218 * @brief Process mouse scroll and adjust zoom accordingly.
219 *
220 * @param yoffset Direction of the scroll (-1, 0, 1).
221 * @param mouse Mouse coordinates ([0,0] in the upper left corner).
222 */
223 void process_mouse_scroll(float yoffset, glm::vec2 mouse)
224 {
225 if (yoffset > -0.0001 && yoffset < 0.0001)
226 return;
227
228 float velocity = -1 * yoffset / 1500.0f * (target_zoom * 100);
229 auto zoom_around = model_mouse_coords(mouse);
230
231 target_zoom += velocity;
232 target_pos =
233 glm::vec3(zoom_around + powf(2.0f, target_zoom * 400) *
234 (glm::vec2(current_pos) - zoom_around) /
235 powf(2.0f, current_zoom * 400),
236 target_pos.z);
237 }
238
239 void process_mouse_button(int button, int action, glm::vec2 mouse_pos)
240 {
241 switch (button) {
242 case GLFW_MOUSE_BUTTON_MIDDLE:
243 switch (action) {
244 case GLFW_PRESS:
245 look_at(model_mouse_coords(mouse_pos));
246 default:
247 break;
248 }
249 default:
250 break;
251 }
252 }
253
254 /**
255 * @brief Convert point coordinates ([0,0] in the upper left corner),
256 * to screen coordinates ([0,0] in the middle of the screen).
257 *
258 * @param point Point coordinates ([0,0] in the upper left corner)
259 * @return glm::vec2 Screen coordinates ([0,0] in the middle of the screen)
260 */
261 glm::vec2 screen_point_coords(glm::vec2 point) const
262 {
263 return glm::vec2(point.x - width / 2.0f, point.y - height / 2.0f);
264 }
265
266 /**
267 * @brief Cause the camera to look at the specified point.
268 *
269 * @param tgt This point will eventually get to the middle of the screen.
270 */
271 void look_at(glm::vec2 tgt)
272 {
273 target_pos.x = tgt.x;
274 target_pos.y = tgt.y;
275 }
276};
