29 /**
30  * @brief A small utility class that manages the viewport coordinates, together
31  * with the virtual "camera" position and zoom.
32  */
33 struct View
34 {
35  using Vector2 = Magnum::Vector2;
36  using Vector2i = Magnum::Vector2i;
37  using Vector3 = Magnum::Vector3;
38  using Matrix3 = Magnum::Matrix3;
39 
40  /** Size of the framebuffer (in screen coordinates). */
41  Vector2i fb_size;
42 
43  /** Current middle point of the view (in model coordinates). */
44  Vector2 mid;
45  /** Intended middle point of the view (camera will move there). */
46  Vector2 mid_target;
47  /** Negative-log2-vertical-size of the on-screen part of the model space (aka zoom). */
48  float view_logv;
49  /** Intended zoom. */
50  float view_logv_target;
51 
52  View()
53  : fb_size(1, 1)
54  , mid(4, 4)
55  , mid_target(4, 4)
56  , view_logv(-4)
57  , view_logv_target(-4)
58  {}
59 
60  /** Utility to convert logv to actual size */
61  inline float zoom_scale(float logv) const { return pow(2, -logv); }
62 
63  /** Return the vertical size of the viewed part in model space. */
64  inline float zoom_scale() const { return zoom_scale(view_logv); }
65 
66  /**
67  * @brief Move the current midpoint and zoom a bit closer to the target midpoint and zoom.
68  *
69  * @param dt Time difference
70  */
71  void update(float dt)
72  {
73  const float ir = pow(0.005, dt);
74  const float r = 1 - ir;
75 
76  view_logv = ir * view_logv + r * view_logv_target;
77  mid = ir * mid + r * mid_target;
78  }
79 
80  /**
81  * @brief Compute and return the view size in model coordinates
82  *
83  * @return Vector2 with the size.
84  */
85  inline Vector2 view_size() const
86  {
87  const float v = zoom_scale();
88  const float h = fb_size.x() * v / fb_size.y();
89  return Vector2(h, v);
90  }
91 
92  /**
93  * @brief Compute the view rectangle coordinates
94  *
95  * @return Tuple of two Vector2, with the lower-left view corner and size of the view.
96  */
97  inline std::tuple<Vector2, Vector2> frame() const
98  {
99  auto s = view_size();
100  return { mid - s / 2, s };
101  }
102 
103  /**
104  * @brief Compute model coordinates from screen coordinates.
105  *
106  * The "zero" point is thought to be in the middle of the screen.
107  *
108  * @param screen Screen coordinates.
109  * @return Model coordinates.
110  */
111  inline Vector2 model_coords(Vector2i screen) const
112  {
113  return mid + view_size() *
114  (Vector2(screen) / Vector2(fb_size) - Vector2(0.5, 0.5));
115  }
116 
117  /**
118  * @brief Computes screen coordinates from model coordinates.
119  *
120  * Inverse to model_coords().
121  *
122  * @param model Model coordinates.
123  * @return Screen coordinates.
124  */
125  inline Vector2 screen_coords(Vector2 model) const
126  {
127  return ((model - mid) / view_size() + Vector2(0.5, 0.5)) *
128  Vector2(fb_size);
129  }
130 
131  /**
132  * @brief Computes screen coordinates of the mouse cursor.
133  *
134  * @param mouse Mouse cursor coordinates ([0, 0] is in the bottom left
135  * corner.)
136  * @return Screen coordinates of the mouse cursor.
137  */
138  inline Vector2i screen_mouse_coords(Vector2i mouse) const
139  {
140  return Vector2i(mouse.x(), fb_size.y() - mouse.y());
141  }
142 
143  /**
144  * @brief Computes model coordinates of the mouse cursor.
145  *
146  * @param mouse Mouse cursor coordinates ([0, 0] is in the bottom left
147  * corner.)
148  * @return Model coordinates of the mouse cursor.
149  */
150  inline Vector2 model_mouse_coords(Vector2i mouse) const
151  {
152  return model_coords(screen_mouse_coords(mouse));
153  }
154 
155  /**
156  * @brief Compute the projection matrix for drawing into the "view" space.
157  *
158  * 1 unit in the view space should be roughly equal to 1 framebuffer pixel,
159  * independently of camera position and zoom.
160  */
161  inline Magnum::Matrix3 screen_projection_matrix() const
162  {
163  return Magnum::Matrix3(Vector3(2.0f / fb_size.x(), 0, 0),
164  Vector3(0, 2.0f / fb_size.y(), 0),
165  Vector3(-1, -1, 1));
166  }
167 
168  /**
169  * @brief Compute the projection matrix for drawing into the "model" coordinates.
170  *
171  * This view is transformed along with camera position an zoom.
172  */
173  inline Magnum::Matrix3 projection_matrix() const
174  {
175  auto isize = 1.0 / view_size();
176 
177  return Magnum::Matrix3(
178  Vector3(2 * isize.x(), 0, 0),
179  Vector3(0, 2 * isize.y(), 0),
180  Vector3(-2 * mid.x() * isize.x(), -2 * mid.y() * isize.y(), 1));
181  }
182 
183  /**
184  * @brief Cause a zoom in response to user action.
185  *
186  * Call this to handle mousewheel scrolling events.
187  */
188  void zoom(float delta, Vector2i mouse)
189  {
190  view_logv_target += delta;
191  if (view_logv_target > 15)
192  view_logv_target = 15;
193  if (view_logv_target < -10)
194  view_logv_target = -10;
195 
196  auto zoom_around = model_mouse_coords(mouse);
197  mid_target = zoom_around + zoom_scale(view_logv_target) *
198  (mid - zoom_around) / zoom_scale();
199  }
200 
201  /**
202  * @brief Cause the camera to look at the specified point.
203  *
204  * @param tgt This point will eventually get to the middle of the screen.
205  */
206  void lookat(Vector2 tgt) { mid_target = tgt; }
207 
208  /** Reset the framebuffer size to the specified value */
209  void set_fb_size(Vector2i s) {
210  fb_size = s;
211  }
212 
213  /** Variant of lookat() that accepts "screen" coordinates. */
214  void lookat_screen(Vector2i mouse) { lookat(model_mouse_coords(mouse)); }
215 };
