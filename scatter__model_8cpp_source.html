27void
28ScatterModel::update(const ScaledData &d,
29 const LandmarkModel &lm,
30 const TrainingConfig &tc)
31{
32 if (dirty(d)) {
33 points.resize(d.n);
34 refresh(d);
35 clean(d);
36 }
37
38 if (lm_watch.dirty(lm)) {
39 refresh(d);
40 lm_watch.clean(lm);
41 }
42
43 const size_t max_points =
44#ifndef ENABLE_CUDA
45 1000
46#else
47 50000
48#endif
49 ;
50
51 auto [ri, rn] = dirty_range(d);
52 if (!rn)
53 return;
54 if (rn > max_points)
55 rn = max_points;
56
57 if (lm.d != d.dim())
58 return;
59
60 clean_range(d, rn);
61
62 auto do_embedsom = [&](size_t from, size_t n) {
63#ifdef ENABLE_CUDA
64 embedsom_cuda.run
65#else
66 embedsom
67#endif
68 (n,
69 lm.n_landmarks(),
70 d.dim(),
71 tc.boost,
72 tc.topn,
73 tc.adjust,
74 d.data.data() + d.dim() * from,
75 lm.hidim_vertices.data(),
76 &lm.lodim_vertices[0][0],
77 &points[from][0]);
78 };
79
80 if (ri + rn >= d.n) {
81 size_t diff = d.n - ri;
82 do_embedsom(ri, diff);
83 ri = 0;
84 rn -= diff;
85 }
86
87 if (rn)
88 do_embedsom(ri, rn);
89}
