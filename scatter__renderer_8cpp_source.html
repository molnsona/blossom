43 ScatterRenderer::draw(const View &view,
44  const ScatterModel &model,
45  const ColorData &colors)
46 {
47  GL::Buffer buffer;
48  size_t n =
49  std::min(model.points.size(),
50  colors.data.size()); // misalignment aborts it, be careful
51 
52  buffer.setData(MeshTools::interleave(
53  Corrade::Containers::ArrayView(model.points.data(), n),
54  Corrade::Containers::ArrayView(colors.data.data(), n)));
55 
56  GL::Renderer::enable(GL::Renderer::Feature::Blending);
57 
58  Magnum::GL::Mesh point_mesh;
59  point_mesh.setPrimitive(MeshPrimitive::Points);
60  point_mesh.setCount(n).addVertexBuffer(std::move(buffer),
61  0,
62  decltype(flat_shader)::Position{},
63  decltype(flat_shader)::Color4{});
64 
65  auto proj = view.projection_matrix();
66 
67  flat_shader.setTransformationProjectionMatrix(proj).draw(point_mesh);
68 
69  GL::Renderer::disable(GL::Renderer::Feature::Blending);
