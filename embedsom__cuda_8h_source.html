118 /** CUDA error code check. This is internal function used by CUCH macro. */
119 inline void
120 _cuda_check(cudaError_t status,
121  int line,
122  const char *srcFile,
123  const char *errMsg = nullptr)
124 {
125  if (status != cudaSuccess) {
126  throw(CudaError((std::stringstream()
127  << "CUDA Error (" << status

129  << "at " << srcFile << "[" << line << "]: " << errMsg)
130  .str(),
131  status));
132  }
