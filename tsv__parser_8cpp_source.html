60  if (values.size() == 0)
61  continue;
62 
63  if (dm.d == 0) {

65  dm.d = values.size();
66  dm.names = values;
67  continue;
68  } else if (dm.d != values.size())
69  throw std::length_error("Row length mismatch");

73  }
