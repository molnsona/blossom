24 void
25 graph_layout_step(GraphLayoutData &data,
26  const MouseData &mouse,
27  LandmarkModel &lm,
28  float time)
29 {
30  auto &vertices = lm.lodim_vertices;
31  const auto &edges = lm.edges;
32  const auto &edge_lengths = lm.edge_lengths;
33 
34  if (time > 0.05)
35  time = 0.05;
36  // check if data is m'kay
37  if (data.velocities.size() != vertices.size())
38  data.velocities.resize(vertices.size(), Magnum::Vector2(0, 0));
39  if (data.forces.size() != vertices.size())
40  data.forces.resize(vertices.size());
41 
42  // clean up the forces
43  for (auto &v : data.forces)
44  v = Magnum::Vector2(0, 0);
45 
46  // add repulsive forces
47  for (size_t i = 1; i < vertices.size(); ++i)
48  for (size_t j = 0; j < i; ++j) {
49  auto d = vertices[j] - vertices[i];
50  auto q = exp(-d.length() * 3) * 1000 * time;
51  data.forces[i] += -q * d;
52  data.forces[j] += q * d;
53  }
54 
55  // add compulsive forces
56  for (size_t i = 0; i < edges.size(); ++i) {
57  auto [p1, p2] = edges[i];
58  auto d = vertices[p2] - vertices[p1];
59  auto q = (edge_lengths[i] - d.length()) * 1000 * time;
60  data.forces[p1] += -q * d;
61  data.forces[p2] += q * d;
62  }
63 
64  // update velocities and positions
65  auto slowdown = pow(0.1f, time);
66  for (size_t i = 0; i < vertices.size(); ++i) {
67  if (mouse.vert_pressed && mouse.vert_ind == i)
68  continue;
69 
70  data.velocities[i] += data.forces[i] * time;
71  vertices[i] += data.velocities[i] * time;
72  data.velocities[i] *= slowdown;
73  }
74 
75  lm.touch();
76 }
