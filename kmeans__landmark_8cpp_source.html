33 void
34 kmeans_landmark_step(KMeansData &data,
35  const ScaledData &model,
36  size_t iters,
37  float alpha,
38  float gravity,
39  LandmarkModel &lm)
40 {
41  size_t n_means = lm.n_landmarks();
42  size_t d = lm.d;
43  auto &means = lm.hidim_vertices;
44 
45  if (!(model.n && n_means))
46  return; // this might happen but it's technically okay
47 
48  if (d != model.dim() || means.size() != n_means * d)
49  return; // TODO throw something useful
50 
51  gravity *= alpha / n_means;
52 
53  std::uniform_int_distribution<size_t> random_target(0, model.n - 1);
54 
55  lm.touch();
56  for (size_t iter = 0; iter < iters; ++iter) {
57  size_t tgt = random_target(data.gen);
58 
59  size_t best = 0;
60  float best_sqdist = std::numeric_limits<float>::infinity();
61 
62  // find the mean that's closest to the target
63  for (size_t mi = 0; mi < n_means; ++mi) {
64  float sqd = 0;
65 
66  for (size_t di = 0; di < d; ++di)
67  sqd += sqr(means[di + d * mi] - model.data[di + d * tgt]);
68 
69  if (sqd < best_sqdist) {
70  best = mi;
71  best_sqdist = sqd;
72  }
73  }
74 
75  // move the means a bit closer
76  for (size_t mi = 0; mi < n_means; ++mi)
77  for (size_t di = 0; di < d; ++di)
78  means[di + d * mi] +=
79  ((mi == best) ? alpha : gravity) *
80  (model.data[di + d * tgt] - means[di + d * mi]);
81  }
82 }

84 void
85 som_landmark_step(KMeansData &data,
86  const ScaledData &model,
87  size_t iters,
88  float alpha,
89  float sigma,
90  LandmarkModel &lm)
91 {
92  size_t n_neurons = lm.n_landmarks();
93  size_t d = lm.d;
94  const auto &map = lm.lodim_vertices;
95  auto &neurons = lm.hidim_vertices;
96 
97  const float nisqsigma = -1.0 / (sigma * sigma);
98 
99  if (!(model.n && n_neurons))
100  return; // this might happen but it's technically okay
101 
102  if (d != model.dim() || neurons.size() != n_neurons * d ||
103  map.size() != n_neurons)
104  return; // TODO throw something useful
105 
106  std::uniform_int_distribution<size_t> random_target(0, model.n - 1);
107 
108  lm.touch();
109  for (size_t iter = 0; iter < iters; ++iter) {
110  size_t tgt = random_target(data.gen);
111 
112  size_t best = 0;
113  float best_sqdist = std::numeric_limits<float>::infinity();
114 
115  // find the mean that's closest to the target
116  for (size_t ni = 0; ni < n_neurons; ++ni) {
117  float sqd = 0;
118 
119  for (size_t di = 0; di < d; ++di)
120  sqd += sqr(neurons[di + d * ni] - model.data[di + d * tgt]);
121 
122  if (sqd < best_sqdist) {
123  best = ni;
124  best_sqdist = sqd;
125  }
126  }
127 
128  // move the rest according to the neighborhood function
129  for (size_t ni = 0; ni < n_neurons; ++ni) {
130  float r = alpha * exp((map[best] - map[ni]).dot() * nisqsigma);
131 
132  for (size_t di = 0; di < d; ++di)
133  neurons[di + d * ni] +=
134  r * (model.data[di + d * tgt] - neurons[di + d * ni]);
135  }
136  }
137 }
