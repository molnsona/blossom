81 void
82 Application::keyPressEvent(KeyEvent &event)
83 {
84  if (ui_imgui.key_press_event(event))
85  return;
86 
87  const float speed = 0.1;
88  switch (event.key()) {
89  case KeyEvent::Key::Left:
90  // all stuff is scaled to vertical size, so the .y() here is OK
91  view.mid_target.x() -= view.view_size().y() * speed;
92  break;
93  case KeyEvent::Key::Right:
94  view.mid_target.x() += view.view_size().y() * speed;
95  break;
96  case KeyEvent::Key::Down:
97  view.mid_target.y() -= view.view_size().y() * speed;
98  break;
99  case KeyEvent::Key::Up:
100  view.mid_target.y() += view.view_size().y() * speed;
101  break;
102  case KeyEvent::Key::LeftCtrl:
103  state.keyboard.ctrl_pressed = true;
104  break;
105  case KeyEvent::Key::RightCtrl:
106  state.keyboard.ctrl_pressed = true;
107  break;
108  }
109  event.setAccepted();
110 }

129 void
130 Application::mousePressEvent(MouseEvent &event)
131 {
132  if (ui_imgui.mouse_press_event(event)) {
133  event.setAccepted(true);
134  return;
135  }
136 
137  // Close menu if it was opened and clicked somewhere else.
138  ui_imgui.menu.close_menu();
139 
140  switch (event.button()) {
141  case MouseEvent::Button::Middle:
142  view.lookat_screen(event.position());
143  return;
144  case MouseEvent::Button::Left:
145  state.mouse.left_pressed = true;
146  state.mouse.mouse_pos = event.position();
147 
148  if (graph_renderer.is_vert_pressed(
149  Vector2(view.screen_mouse_coords(state.mouse.mouse_pos)),
150  state.mouse.vert_ind)) {
151  if (state.keyboard.ctrl_pressed) {
152  state.landmarks.duplicate(state.mouse.vert_ind);
153  } else {
154  state.mouse.vert_pressed = true;
155  state.landmarks.move(
156  state.mouse.vert_ind,
157  view.model_mouse_coords(state.mouse.mouse_pos));
158  }
159  } else if (state.keyboard.ctrl_pressed) {
160  state.landmarks.add(
161  view.model_mouse_coords(state.mouse.mouse_pos));
162  }
163 
164  break;
165  case MouseEvent::Button::Right:
166  state.mouse.right_pressed = true;
167  state.mouse.mouse_pos = event.position();
168 
169  if (graph_renderer.is_vert_pressed(
170  Vector2(view.screen_mouse_coords(state.mouse.mouse_pos)),
171  state.mouse.vert_ind)) {
172  if (state.keyboard.ctrl_pressed) {
173  state.landmarks.remove(state.mouse.vert_ind);
174  }
175  }
176  break;
177  }
178 }
