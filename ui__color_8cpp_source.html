31void
32UiColorSettings::render(State &state, ImGuiWindowFlags window_flags)
33{
34 if (!show_window)
35 return;
36
37 auto column_combo = [&](const std::string &combo_name, int &column_ind) {
38 ImGui::Text("Column for colors:");
39 auto dim = state.data.names.size();
40 if (!dim) {
41 ImGui::Text("No columns are present.");
42 return;
43 }
44
45 if (ImGui::BeginCombo(combo_name.data(),
46 state.data.names[column_ind].c_str())) {
47 bool ret = false;
48 for (size_t i = 0; i < dim; ++i) {
49 bool is_selected = (int)i == column_ind;
50 if (ImGui::Selectable(state.data.names[i].c_str(),
51 &is_selected)) {
52 column_ind = i;
53 ret = true;
54 }
55 }
56
57 if (ret)
58 state.colors.touch_config();
59 ImGui::EndCombo();
60 }
61 };
62
63 if (ImGui::Begin("Color", &show_window, window_flags)) {
64 ImGui::Text("Style of coloring:");
65
66 if (reset_button()) {
67 state.colors.reset();
68 }
69
70 if (ImGui::RadioButton("Expression",
71 &state.colors.coloring,
72 (int)ColorData::Coloring::EXPR))
73 state.colors.touch_config();
74
75 if (ImGui::RadioButton("Discretized clusters",
76 &state.colors.coloring,
77 (int)ColorData::Coloring::CLUSTER))
78 state.colors.touch_config();
79
80 if (ImGui::SliderFloat("Alpha##color",
81 &state.colors.alpha,
82 0.0f,
83 1.0f,
84 "%.3f",
85 ImGuiSliderFlags_AlwaysClamp))
86 state.colors.touch_config();
87
88 ImGui::Separator();
89
90 switch (state.colors.coloring) {
91 case (int)ColorData::Coloring::EXPR: {
92 column_combo("##columnsexpr", state.colors.expr_col);
93
94 ImGui::Text("Color palette:");
95 if (ImGui::BeginCombo("##palettes",
96 state.colors.col_palette.c_str())) {
97 bool ret = false;
98 for (auto &[name, _] : colormap::palettes) {
99 bool is_selected = name == state.colors.col_palette;
100 if (ImGui::Selectable(name.c_str(), &is_selected)) {
101 state.colors.col_palette = name;
102 ret = true;
103 }
104 }
105
106 if (ret)
107 state.colors.touch_config();
108 ImGui::EndCombo();
109 }
110
111 if (ImGui::Checkbox("Reverse color palette",
112 &state.colors.reverse))
113 state.colors.touch_config();
114 } break;
115 case int(ColorData::Coloring::CLUSTER):
116 column_combo("##columnscluster", state.colors.cluster_col);
117
118 if (ImGui::SliderInt(
119 "Cluster count", &state.colors.cluster_cnt, 1, 50))
120 state.colors.touch_config();
121
122 break;
123 }
124
125 ImGui::End();
126 }
127}
