24 void
25 ScaledData::update(const TransData &td)
26 {
27  if (dirty(td) && (td.dim() != dim() || td.n != n)) {
28  n = td.n;
29  config.resize(td.dim());
30  touch_config();
31  data.resize(n * dim());
32  clean(td);
33  }
34 
35  auto [ri, rn] = dirty_range(td);
36  if (!rn)
37  return;
38  if (rn > 5000)
39  rn = 5000;
40  clean_range(td, rn);
41 
42  std::vector<float> means = td.sums;
43  std::vector<float> isds = td.sqsums;
44  size_t d = dim();
45  for (size_t di = 0; di < d; ++di) {
46  means[di] /= n;
47  isds[di] /= n;
48  isds[di] = 1 / sqrt(isds[di] - means[di] * means[di]);
49  if (isds[di] > 10000)
50  isds[di] = 10000;
51  }
52 
53  for (; rn-- > 0; ++ri) {
54  if (ri >= n)
55  ri = 0;
56  for (size_t di = 0; di < d; ++di)
57  data[ri * d + di] =
58  (td.data[ri * d + di] - means[di]) *
59  (config[di].scale ? config[di].sdev * isds[di] : 1);
60  }
61  touch();
62 }
