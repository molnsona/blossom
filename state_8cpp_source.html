25void
26State::update(float actual_time, bool vert_pressed, int vert_ind)
27{
28 // avoid simulation explosions on long frames
29 float time = actual_time;
30 if (time > 0.05)
31 time = 0.05;
32
33 // Compute time for estimation batch sizes computations.
34 frame_stats.update_times();
35
36 stats.update(data);
37
38 trans.update(data, stats, frame_stats);
39 scaled.update(trans, frame_stats);
40
41 // TODO only run this on data reset, ideally from trans or from a common
42 // trigger
43#ifdef STATE_DEBUG // TODO remove
44 landmarks.update_dim(3);
45#else
46 landmarks.update_dim(scaled.dim());
47#endif
48
49 if (training_conf.kmeans_landmark)
50 kmeans_landmark_step(kmeans_data,
51 scaled,
52 training_conf.kmeans_iters,
53 training_conf.kmeans_alpha,
54 training_conf.gravity,
55 landmarks);
56
57 if (training_conf.knn_edges)
58 make_knn_edges(knn_data,
59 landmarks,
60 training_conf.kns); // TODO: Edges are not removed, once
61 // they are created.
62
63 if (training_conf.graph_layout)
64 graph_layout_step(layout_data, vert_pressed, vert_ind, landmarks, time);
65
66 if (training_conf.tsne_layout)
67 tsne_layout_step(tsne_data, vert_pressed, vert_ind, landmarks, time);
68
69 if (training_conf.som_landmark)
70 som_landmark_step(kmeans_data,
71 scaled,
72 training_conf.som_iters,
73 training_conf.som_alpha,
74 training_conf.sigma,
75 landmarks);
76
77 colors.update(trans, landmarks, frame_stats);
78 scatter.update(scaled, landmarks, training_conf, frame_stats);
79}
