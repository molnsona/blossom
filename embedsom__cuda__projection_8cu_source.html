103 template<typename F, typename ArrayF>
104 __inline__ __device__ void
105 readAligned(F *const __restrict__ dst,
106  const F *const __restrict__ src,
107  const knn_entry<F> *const __restrict__ neighbors,
108  const uint32_t n,
109  const uint32_t dim,
110  const uint32_t groupRank,
111  const uint32_t groupSize)
112 {

114 
115  const uint32_t loadsCount = dim / size;
116  const uint32_t dimX = dim / size;
117

119  const ArrayF *const srcX = reinterpret_cast<const ArrayF *>(src);
120 
121  for (size_t i = groupRank; i < n * loadsCount; i += groupSize) {
122  const auto idx = i / loadsCount;

124 
125  dstX[idx * dimX + off] = srcX[neighbors[idx].index * dimX + off];

181 /** Compute scores for a vector of sorted distances */
182 template<typename F>
183 __inline__ __device__ void
184 sortedDistsToScores(knn_entry<F> *const __restrict__ neighbors,
185  const std::size_t adjustedK,
186  const std::size_t k,
187  const F boost)
188 {
189  // compute the distance distribution for the scores
190  F mean = 0, sd = 0, wsum = 0;
191  for (uint32_t i = 0; i < adjustedK; ++i) {
192  const F tmp = sqrt(neighbors[i].distance);
193  const F w = 1 / F(i + 1);
194  mean += tmp * w;
195  sd += tmp * tmp * w;
196  wsum += w;
197  neighbors[i].distance = tmp;
198  }
199 
200  mean /= wsum;
201  sd = boost / sqrt(sd / wsum - mean * mean);
202  const F nmax =
203  EmbedSOMConstants<F>::maxAvoidance / neighbors[adjustedK - 1].distance;
204 
205  // convert the stuff to scores
206  for (uint32_t i = 0; i < k; ++i) {
207  if (k < adjustedK)
208  neighbors[i].distance =
209  exp((mean - neighbors[i].distance) * sd) *
210  (1 - exp(neighbors[i].distance * nmax -
211  EmbedSOMConstants<F>::maxAvoidance));
212  else
213  neighbors[i].distance = exp((mean - neighbors[i].distance) * sd);
214  }
215 }

217 /** Convert the distances to scores using thread_block_tile and its built in
218  * functions for reduction and shuffle.
219  */
220 template<typename F, class SCORE_STORAGE, class TILE>
221 __inline__ __device__ void
222 sortedDistsToScoresGroup(const TILE &tile,
223  SCORE_STORAGE storage,
224  const std::size_t adjustedK,
225  const std::size_t k,
226  const F boost)
227 {
228  // TRICKY: if k is big enough and tile is small enough, this array may
229  // overflow. Should be MAX_K / tile.size()
230  F tmpScores[10];
231  F lastScore;
232 
233  // compute the distance distribution for the scores
234  F mean = 0, sd = 0, wsum = 0;
235  for (uint32_t i = tile.thread_rank(); i < adjustedK; i += tile.size()) {
236  const F tmp = sqrt(storage.getNeighborDistance(i));
237  const F w = 1 / F(i + 1);
238  mean += tmp * w;
239  wsum += w;
240  tmpScores[i / tile.size()] = tmp;
241  }
242 
243  {
244  mean = cg::reduce(tile, mean, cg::plus<F>());
245  wsum = cg::reduce(tile, wsum, cg::plus<F>());
246  }
247 
248  mean /= wsum;
249 
250  for (uint32_t i = tile.thread_rank(); i < adjustedK; i += tile.size()) {
251  const F tmp = tmpScores[i / tile.size()] - mean;
252  const F w = 1 / F(i + 1);
253  sd += tmp * tmp * w;
254  }
255 
256  {
257  sd = cg::reduce(tile, sd, cg::plus<F>());
258 
259  const auto lastScoreThreadIdx = (adjustedK - 1) % tile.size();
260  const auto lastScoreIdx = (adjustedK - 1) / tile.size();
261  lastScore = tile.shfl(tmpScores[lastScoreIdx], lastScoreThreadIdx);
262  }
263 
264  sd = boost / sqrt(sd / wsum);
265  const F nmax = EmbedSOMConstants<F>::maxAvoidance / lastScore;
266 
267  // convert the stuff to scores
268  if (k < adjustedK)
269  for (uint32_t i = tile.thread_rank(); i < k; i += tile.size()) {
270  const auto scoreIdx = i / tile.size();
271  const F score = exp((mean - tmpScores[scoreIdx]) * sd) *
272  (1 - exp(tmpScores[scoreIdx] * nmax -
273  EmbedSOMConstants<F>::maxAvoidance));
274  storage.storeScore(i, score);
275  }
276  else
277  for (uint32_t i = tile.thread_rank(); i < k; i += tile.size())
278  storage.storeScore(i,
279  exp((mean - tmpScores[i / tile.size()]) * sd));
280 }

283 template<typename F>

285 addGravity(const F score,

315 /** Compute a projection of the point to a line defined by points I and J */
316 template<typename F>
317 __inline__ __device__ typename Vec<2, F>::Type
318 euclideanProjection(const F *const __restrict__ point,
319  const F *const __restrict__ gridPointI,
320  const F *const __restrict__ gridPointJ,
321  const uint32_t dim)
322 {
323  typename Vec<2, F>::Type result{ 0.0, 0.0 };
324  for (uint32_t k = 0; k < dim; ++k) {
325  const F tmp = gridPointJ[k] - gridPointI[k];
326  result.y += tmp * tmp;
327  result.x += tmp * (point[k] - gridPointI[k]);
328  }
329  return result;
330 }

332 /** Run the projection on 4-vectors */
333 template<typename F>
334 __inline__ __device__ typename Vec<2, F>::Type
335 euclideanProjection4Wise(const F *const __restrict__ point,
336  const F *const __restrict__ gridPointI,
337  const F *const __restrict__ gridPointJ,
338  const uint32_t dim)
339 {
340  const auto *const __restrict__ gridPointI4 =
341  reinterpret_cast<const typename Vec<4, F>::Type *>(gridPointI);
342  const auto *const __restrict__ gridPointJ4 =
343  reinterpret_cast<const typename Vec<4, F>::Type *>(gridPointJ);
344  const auto *const __restrict__ point4 =
345  reinterpret_cast<const typename Vec<4, F>::Type *>(point);
346 
347  typename Vec<2, F>::Type result{ 0.0, 0.0 };
348 
349 #define DOIT(X) \
350  tmp = tmpGridJ.X - tmpGridI.X; \
351  result.y += tmp * tmp; \
352  result.x += tmp * (tmpPoint.X - tmpGridI.X)
353 
354  for (uint32_t k = 0; k < dim / 4; ++k) {
355  const auto tmpGridI = gridPointI4[k];
356  const auto tmpGridJ = gridPointJ4[k];
357  const auto tmpPoint = point4[k];
358 
359  F tmp;
360  DOIT(x);
361  DOIT(y);
362  DOIT(z);
363  DOIT(w);
364  }
365 #undef DOIT
366 
367  for (uint32_t k = dim - (dim % 4); k < dim; ++k) {
368  const F tmp = gridPointJ[k] - gridPointI[k];
369  result.y += tmp * tmp;
370  result.x += tmp * (point[k] - gridPointI[k]);
371  }
372 
373  return result;
374 }

376 /** Add the result of projections to the approximation matrix */
377 template<typename F>
378 __inline__ __device__ void
379 addApproximation(const F scoreI,
380  const F scoreJ,
381  const F *const __restrict__ grid2DPointI,
382  const F *const __restrict__ grid2DPointJ,
383  const F adjust,
384  const F scalarProjection,
385  F *const __restrict__ mtx)
386 {
387  F h[2], hp = 0;
388 #pragma unroll
389  for (uint32_t i = 0; i < 2; ++i) {
390  h[i] = grid2DPointJ[i] - grid2DPointI[i];
391  hp += h[i] * h[i];
392  }
393 
394  if (hp < EmbedSOMConstants<F>::zeroAvoidance)
395  return;
396 
397  const F exponent = scalarProjection - .5;
398  const F s =
399  scoreI * scoreJ * pow(1 + hp, adjust) * exp(-exponent * exponent);
400  const F sihp = s / hp;
401  const F rhsc = s * (scalarProjection +
402  (h[0] * grid2DPointI[0] + h[1] * grid2DPointI[1]) / hp);
403 
404  mtx[0] += h[0] * h[0] * sihp;
405  mtx[1] += h[0] * h[1] * sihp;
406  mtx[2] += h[1] * h[0] * sihp;
407  mtx[3] += h[1] * h[1] * sihp;
408  mtx[4] += h[0] * rhsc;
409  mtx[5] += h[1] * rhsc;
410 }

412 /** Same as addApproximation(), on 2-vectors. */
413 template<typename F>
414 __inline__ __device__ void
415 addApproximation2Wise(const F scoreI,
416  const F scoreJ,
417  const F *const __restrict__ grid2DPointI,
418  const F *const __restrict__ grid2DPointJ,
419  const F adjust,
420  const F scalarProjection,
421  F *const __restrict__ mtx)
422 {
423  const typename Vec<2, F>::Type tmpGrid2dI =
424  reinterpret_cast<const typename Vec<2, F>::Type *>(grid2DPointI)[0];
425  const typename Vec<2, F>::Type tmpGrid2dJ =
426  reinterpret_cast<const typename Vec<2, F>::Type *>(grid2DPointJ)[0];
427 
428  const F h[2]{ tmpGrid2dJ.x - tmpGrid2dI.x, tmpGrid2dJ.y - tmpGrid2dI.y };
429  const F hp = h[0] * h[0] + h[1] * h[1];
430 
431  if (hp < EmbedSOMConstants<F>::zeroAvoidance)
432  return;
433 
434  const F exponent = scalarProjection - .5;
435  const F s =
436  scoreI * scoreJ * pow(1 + hp, adjust) * exp(-exponent * exponent);
437  const F sihp = s / hp;
438  const F rhsc =
439  s * (scalarProjection + (h[0] * tmpGrid2dI.x + h[1] * tmpGrid2dI.y) / hp);
440 
441  mtx[0] += h[0] * h[0] * sihp;
442  mtx[1] += h[0] * h[1] * sihp;
443  mtx[2] += h[1] * h[0] * sihp;
444  mtx[3] += h[1] * h[1] * sihp;
445  mtx[4] += h[0] * rhsc;
446  mtx[5] += h[1] * rhsc;

472 /** Specialization of getIndices for the rectangle indexing. */
473 template<>
474 __inline__ __device__ uint2
475 getIndices<RectangleIndexer>(uint32_t plainIndex, uint32_t k)
476 {

486 /** Base projection kernel (each thread does everything for a single point). */
487 template<typename F>
488 __global__ void
489 projectionBaseKernel(const F *__restrict__ points,
490  const F *const __restrict__ grid,
491  const F *const __restrict__ grid2d,
492  knn_entry<F> *__restrict__ neighbors,
493  F *__restrict__ projections,
494  const uint32_t dim,
495  const uint32_t n,
496  const uint32_t gridSize,
497  const uint32_t k,
498  const F adjust,
499  const F boost)
500 {
501  // assign defaults and generate scores
502  {
503  const uint32_t adjustedK = k < gridSize ? k + 1 : k;
504  const uint32_t pointIdx = blockIdx.x * blockDim.x + threadIdx.x;
505  if (pointIdx >= n)
506  return;
507  points = points + pointIdx * dim;
508  neighbors = neighbors + pointIdx * adjustedK;
509  projections = projections + pointIdx * 2;
510  sortedDistsToScores<F>(neighbors, adjustedK, k, boost);
511  }
512 
513  F mtx[6];
514  memset(mtx, 0, 6 * sizeof(F));
515  for (uint32_t i = 0; i < k; ++i) {
516  const uint32_t idxI = neighbors[i].index;
517  const F scoreI = neighbors[i].distance;
518  addGravity(scoreI, grid2d + idxI * 2, mtx);
519  for (uint32_t j = i + 1; j < k; ++j) {
520  const uint32_t idxJ = neighbors[j].index;
521  const F scoreJ = neighbors[j].distance;
522  const auto result = euclideanProjection<F>(
523  points, grid + idxI * dim, grid + idxJ * dim, dim);
524  F scalarProjection = result.x;
525  const F squaredGridPointsDistance = result.y;
526  if (squaredGridPointsDistance == F(0))
527  continue;
528  scalarProjection /= squaredGridPointsDistance;
529  addApproximation(scoreI,
530  scoreJ,
531  grid2d + idxI * 2,
532  grid2d + idxJ * 2,
533  adjust,
534  scalarProjection,
535  mtx);
536  }
537  }
538  // solve linear equation
539  const F det = mtx[0] * mtx[3] - mtx[1] * mtx[2];
540  projections[0] = (mtx[4] * mtx[3] - mtx[5] * mtx[2]) / det;
541  projections[1] = (mtx[0] * mtx[5] - mtx[1] * mtx[4]) / det;
542 }

544 /** Aligned-shared-memory projection kernel.
545  *
546  * One block computes embedding for one point, using CUB block reduce for
547  * matrix reduction.
548  */
549 template<typename F, typename INDEXER, size_t tileSize>
550 __global__ void
551 projectionAlignedShMemoryKernel(const F *__restrict__ points,
552  const F *const __restrict__ grid,
553  const F *const __restrict__ grid2d,
554  knn_entry<F> *__restrict__ neighbors,
555  F *__restrict__ projections,
556  const uint32_t dim,
557  const uint32_t n,
558  const uint32_t gridSize,
559  const uint32_t k,
560  const F adjust,
561  const F boost,
562  const uint32_t groupSize,
563  const uint32_t cacheLeadingDim)
564 {
565  extern __shared__ char sharedMemory[];
566 
567  const uint32_t groupRank = threadIdx.x % groupSize;
568  const uint32_t groupIdx = threadIdx.x / groupSize;
569  const uint32_t groupsCount = blockDim.x / groupSize;
570 
571  const auto grid2dPadding =
572  (k * 3) % cacheLeadingDim == 0
573  ? 0
574  : cacheLeadingDim - ((k * 3) % cacheLeadingDim);
575  auto sharedMemoryoff =
576  reinterpret_cast<F *>(sharedMemory) +
577  ((k + 1) * cacheLeadingDim + k * 3 + grid2dPadding) * groupIdx;
578 
579  F *const __restrict__ pointCache = sharedMemoryoff;
580  F *const __restrict__ gridCache = sharedMemoryoff + cacheLeadingDim;
581  F *const __restrict__ grid2dCache =
582  sharedMemoryoff + (k + 1) * cacheLeadingDim;
583  F *const __restrict__ scoreCache = grid2dCache + k * 2;
584 
585  F *const __restrict__ reduceFinishStorage =
586  reinterpret_cast<F *>(sharedMemory) +
587  ((k + 1) * cacheLeadingDim + k * 3 + grid2dPadding) * groupsCount;
588 
589  auto tile = cg::tiled_partition<tileSize>(cg::this_thread_block());
590 
591  // assign defaults and generate scores
592  {
593  const uint32_t adjustedK = k < gridSize ? k + 1 : k;
594 
595  const auto workIdx = blockIdx.x * groupsCount + groupIdx;
596 
597  if (workIdx >= n)
598  return;
599 
600  points = points + workIdx * dim;
601  neighbors = neighbors + workIdx * adjustedK;
602  projections = projections + workIdx * 2;
603 
604  if (groupRank < tile.size())
605  sortedDistsToScoresGroup<F>(
606  tile,
607  NeighborScoreStorage<F>{ neighbors, scoreCache },
608  adjustedK,
609  k,
610  boost);
611  else
612  storeToCache(groupRank - tile.size(),
613  groupSize - tile.size(),
614  points,
615  pointCache,
616  grid,
617  gridCache,
618  grid2d,
619  grid2dCache,
620  neighbors,
621  dim,
622  cacheLeadingDim,
623  k);
624 
625  if (groupSize == tile.size())
626  storeToCache(groupRank,
627  groupSize,
628  points,
629  pointCache,
630  grid,
631  gridCache,
632  grid2d,
633  grid2dCache,
634  neighbors,
635  dim,
636  cacheLeadingDim,
637  k);
638 
639  __syncthreads();
640  }
641 
642  F mtx[6];
643  memset(mtx, 0, 6 * sizeof(F));
644 
645  for (uint32_t i = groupRank; i < k; i += groupSize)
646  addGravity2Wise(scoreCache[i], grid2dCache + i * 2, mtx);
647 
648  const uint32_t neighborPairs = (k * (k - 1)) / 2;
